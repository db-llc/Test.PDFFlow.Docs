Quick Start

Brief

The article describes how to quickly create your first PDF document.

Details

To create a new document,

  • Add a PDF Flow Library to the project.

  • Call method New to Create a Document.

  • AddSection or several sections to the document.

    You can also quickly AddDefaultSection with the default settings (margins 50, paper size A4, portrait orientation and Arabic numeration style)

    See Configuring section for more information.

  • Add content to every section.

    • Add Paragraph.

    • Add Image.

    • Add Table

    • Add Multilevel List.

    • Add Repeating Areas to the section to show headers and/or footers.

    • Add Line

    • Add StackFlow

  • Create Outline for the document.

  • Call Build with a destination fileName or Build with a stream as a parameter to generate a document.

See Creating Document for more information.

    // New document:
 
    DocumentBuilder builder = DocumentBuilder.New();
 
    // Add section:
 
    // Default configuration (margins 50, paper size A4, portrait orientation and arabic numeration style):
    Section s = builder.AddDefaultSection()
    // Customize settings:
                .SetMargins(20).SetSize(PaperSize.A5).SetOrientation(PageOrientation.Landscape).SetNumerationStyle(NumerationStyle.UpperRoman);
 
    // Add paragraph:
 
    s.AddParagraph("Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua")
        .SetAlignment(HorizontalAlignment.Left).SetFirstLineIndent(20).SetJustifyAlignment(true).SetMargins(10)
        .SetFont(Font.Courier(16)).SetFontColor(Color.Red).SetBold().SetBackColor(Color.Green)
        .SetBorderStyle(Stroke.Solid).SetBorderWidth(3).SetBorderColor(Color.FromRgba(0.6, 0.4, 0.7))
        .SetIsOutline().SetOutlineLevel(0)
        .AddText("Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua")
        .SetBackgroundColor(Color.Yellow).SetStrikethrough(Stroke.Solid, Color.Blue);
 
    // Add tabulation:
 
    s.AddParagraph().AddTabSymbol().AddTextToParagraph("RIGHT300").AddTabulation(300, TabulationType.Right);
 
    // Add page number:
 
    s.AddParagraph("Page #").AddPageNumber().AddTextToParagraph(" | ").AddText("My first document").SetBold();
 
    // Add image:
 
    s.AddImage(imageUrl).SetWidth(200).SetHeight(100).SetScale(ScalingMode.Normal).SetBorderStyle(Stroke.Dotted);
 
    // Add inline image:
 
    s.AddParagraph("Inline ").SetAlignment(HorizontalAlignment.Left).AddInlineImage(imageExclamationMark, ScalingMode.Stretch, 15, 15).AddText(" image");
 
    // Add URL:
 
    // Underlined and Blue color by default:
    s.AddParagraph().AddUrl("http://www.gehtsoftusa.com/", "GehtSoft LLC.");
    // Customize:
    s.AddParagraph().AddUrl("http://www.gehtsoftusa.com/").SetFontColor(Color.Green).SetUnderlineColor(Color.Red);
 
    // Add table:
 
    Table table = s.AddTable();
    table.SetName("RowSpan").SetAlignment(HorizontalAlignment.Center).SetWidth(300)
       .ContentRowStyle.SetHorizontalAlignment(HorizontalAlignment.Center).SetVerticalAlignment(VerticalAlignment.Center);
    table.AddRow().AddCellToRow("England", 0, 3).AddCellToRow("Jaguar").AddCell("12.000");
    table.AddRow().AddCellToRow().AddCellToRow("Jensen").AddCell("24.000");
    table.AddRow().AddCellToRow().AddCellToRow("Triumph").AddCell("20.000");
    table.AddRow().AddCell("Total cars - 3 ", 2).SetHorizontalAlignment(HorizontalAlignment.Right);
 
    // Add MultiLevel list:
 
    s.AddParagraph("Level = 0. Item 1.").SetAlignment(HorizontalAlignment.Left).SetFont(font).SetListBulleted(ListBullet.Bullet, 0, leftIndent);
    s.AddParagraph("Level = 0. Item 2.").SetAlignment(HorizontalAlignment.Left).SetFont(font).SetListBulleted(ListBullet.Bullet, 0, leftIndent);
    s.AddParagraph("Level = 1. Item 1.").SetFont(font).SetFontSize(16).SetListNumbered(NumerationStyle.Arabic, 1, leftIndent);
    s.AddParagraph("Level = 2. Item 1.").SetFont(font).SetFontSize(12).SetListNumbered(NumerationStyle.UpperLatin, 2, leftIndent);
    s.AddParagraph("Level = 2. Item 2.").SetFont(font).SetFontSize(12).SetListNumbered(NumerationStyle.UpperLatin, 2, leftIndent);
    s.AddParagraph("Level = 1. Item 2.").SetAlignment(HorizontalAlignment.Left).SetFont(font).SetFontSize(16).SetListNumbered(NumerationStyle.Arabic, 1, leftIndent);
    s.AddParagraph("Level = 0. Item 3.").SetAlignment(HorizontalAlignment.Left).SetFont(font).SetListBulleted(ListBullet.Bullet, 0, leftIndent);
 
    // Add line:
 
    s.AddLine().SetWidth(1f).SetLength(300).SetColor(Color.Blue).SetStyle(Stroke.Dashed).SetMargins(top: 0, left: 0, right: 0, bottom: 30);
    s.AddLine().SetWidth(5f).SetColor(Color.Red).SetStyle(Stroke.Dotted);
 
    // Build a document:
 
    builder.Build("QuickStart.pdf");

var file = Path.Combine(folderPath, "fileName.pdf");
if (File.Exists(file))
   File.Delete(file);
 
   // New document:
 
   DocumentBuilder.New()
 
   // Add first section:
 
   .AddSection(s =>
   {
 
       // Configure section:
 
       s.SetMargins(20);
       s.SetSize(() => PaperSize.A4);
       s.SetOrientation(() => PageOrientation.Portrait);
       s.SetNumerationStyle(NumerationStyle.Arabic);
 
       // Add paragraph:
 
       s.AddParagraph(p =>
       {
            p.AddText("Some text");
            p.SetFont(new Font {Name = "Courier", Size = 16f, Color = Color.White});
            p.Alignment = HorizontalAlignment.Center;
            p.BackgroundColor = Color.Blue;
            p.SetBorder(b =>
            {
                b.SetWidth(3);
                b.SetStyle(Stroke.Solid);
                b.SetColor(Color.FromRgba(0.6, 0.4, 0.7));
            });
            p.IsOutline = true;
            p.OutlineLevel = 0;
       });
 
       // Add paragraph with inline image:
 
       s.AddParagraph(p =>
       {
            p.Alignment = HorizontalAlignment.Left;
            p.AddContent(() => GenerateTextElements(1, 1, 3));
            p.AddInlineImage(image =>
            {
                image.Height = 15;
                image.Width = 15;
                image.SetFile(inlineImagePath);
                image.Scale = ScalingMode.Stretch;
            });
            p.AddContent(() => GenerateTextElements(1, 1, 3));
       });
 
       // Add image:
 
       s.AddImage(imageUrl, image =>
       {
            image.Width = 1000;
            image.SetMargins(10, 10, 0, 0);
            image.SetScale(ScalingMode.Stretch);
            image.SetBorder(b => b.SetStyle(Stroke.Dotted).SetColor(Color.Black));
       });
 
       // Add URL:
 
       s.AddParagraph(p =>
       {
            p.AddContent(() => new List<ParagraphElement>
            {
                new Url
                {
                    DisplayText = "GehtSoft LLC.",
                    Value = "http://www.gehtsoftusa.com/",
                    Font = new Font
                    {
                        Name = FontNames.Courier,
                        Bold = true,
                        Size = 16,
                        Color = Color.Green,
                        Underline = Stroke.Solid,
                        UnderlineColor = Color.Red
                    }
                }
            });
       });
   })

   .Build(file);
